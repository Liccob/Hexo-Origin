## 二叉树遍历

### 前序遍历

顺序是左子节点 -----> 父节点 -----> 右子节点

```javascript
let res = [];
function digui(pRoot){
    if(!pRoot) return 
    res.push(pRoot);
    digui(pRoot.left);
    digui(pRoot.right);
}

let res2 = [];
function xunhuan(pRoot){
    const stack = [];
    while(!!pRoot || stack.length!=0){
      if(!!pRoot){
        res2.push(pRoot);
        pRoot = pRoot.left
      }else{
        pRoot = pRoot.right
      }
    }
}

```
